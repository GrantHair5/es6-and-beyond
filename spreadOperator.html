Spread Operator

The spread operator basically lets you smoosh two objects or arrays together

It is used with the ... syntax
You can try the following commands:

                        const demoArraySpread = () => {
                            const myArrayOne = [
                                {
                                    name: "Alice",
                                    age: 28
                                },
                                {
                                    name: "Bob",
                                    age: 23
                                }
                            ]
                        
                            const myArrayTwo = [
                                {
                                    name: "Charles",
                                    age: 47
                                },
                                {
                                    name: "David",
                                    age: 64
                                }
                            ]
                        
                            const mySquashedArray = [...myArrayOne, ...myArrayTwo]
                            console.log(JSON.stringify(mySquashedArray))
                        }

                    

                        const demoObjectSpread = () => {
                        const myObjectOne =
                        {
                            Id: 1,
                            Price: 100
                    
                        }
                    
                        const myObjectTwo =
                        {
                            Id: 2,
                            Price: 5000000
                        }
                        const myObjectThree =
                        {
                            Description: "This is an expensive item"
                        }
                    
                    
                        const mySquashedObject = { ...myObjectOne, ...myObjectTwo, ...myObjectThree }
                        console.log(JSON.stringify(mySquashedObject))
                    }
