Rest parameters

Rest parameters allow you to pass multiple params into a function as an array

It is used with the ... syntax similar to spread but it does a different job all together
You can try the following commands:

                        const theOldWay = (a,b,c,d,e,f,g,h,i,j,k,l,m,n,o,p,q,r,s,t,u,v,w,x,y,z) => {
                            console.log("a: ", a)
                            console.log("b: ", b)
                            console.log("c: ", c)
                            //... And so on
                        
                            console.log("z: ", z)
                        }
                        
                        theOldWay("a","b","c","d","e","f","g","h","i","j","k","l","m","n","o","p","q","r","s","t","u","v","w","x","y","z")

                    

                        const theRestParameterWay = (a, ...restOfAlphabet) => {
                            console.log("theRestParameterWay")
                            console.log("a:", a)
                            restOfAlphabet.forEach(letter => {
                                console.log(`${letter}: ${letter}`)
                            })
                        }
